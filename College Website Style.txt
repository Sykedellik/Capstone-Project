                    data.forEach(subject => {
                        const timeString = `${new Date('1970-01-01T' + subject.time_start).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})} - ${new Date('1970-01-01T' + subject.time_end).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}`;
                        let meetLinkHtml = '<span class="text-gray-400">N/A</span>';
                        if (subject.google_meet_link) {
                            meetLinkHtml = `<a href="${subject.google_meet_link}" target="_blank" class="text-green-600 hover:text-green-800" title="Join Meeting">JOIN CLASS <i class="fa-solid fa-video"></i></a>`;
                        }

                        // --- 2. Each piece of data is now in its own <td> cell ---
                        // This works perfectly for both desktop columns and mobile card rows.
                        tableHTML += `<tr class="border-b">
                            <td class="py-3 px-3 font-mono" data-label="Code">${subject.subject_code}</td>
                            <td class="py-3 px-3 font-semibold" data-label="Subject Name">${subject.subject_name}</td>
                            <td class="py-3 px-3" data-label="Day">${subject.day}</td>
                            <td class="py-3 px-3" data-label="Time">${timeString}</td>
                            <td class="py-3 px-3" data-label="Instructor">${subject.instructor_full_name || 'TBA'}</td>
                            <td class="py-3 px-3" data-label="Email">${subject.instructor_email || 'N/A'}</td>
                            <td class="py-3 px-3 text-center" data-label="Meet Link">${meetLinkHtml}</td>
                        </tr>`;
                    });
                    scheduleModalContent.innerHTML = tableHTML + '</tbody></table></div>';
                }).catch(error => {
                    console.error('Fetch error:', error);
                });
        });
    });
    if (scheduleModalCloseBtn) { scheduleModalCloseBtn.addEventListener('click', () => scheduleModal.classList.add('hidden')); }

    // ACCORDION SCRIPT
    document.querySelectorAll('.accordion-header').forEach(button => {
        button.addEventListener('click', () => {
            button.nextElementSibling.classList.toggle('hidden');
            button.querySelector('i.fa-chevron-down').classList.toggle('rotate-180');
        });
    });
});
