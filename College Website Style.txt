    <script>
      const signInStaffButton = document.getElementById(&#39;signInStaff&#39;);
        const signInStudentButton = document.getElementById(&#39;signInStudent&#39;);
        const container = document.getElementById(&#39;container&#39;);

        signInStaffButton.addEventListener(&#39;click&#39;, () =&gt; {
            container.classList.add(&#34;right-panel-active&#34;);
        });

        signInStudentButton.addEventListener(&#39;click&#39;, () =&gt; {
            container.classList.remove(&#34;right-panel-active&#34;);
        });

        document.addEventListener(&#39;DOMContentLoaded&#39;, function () {
            const passwordInput = document.getElementById(&#39;password&#39;);
            const togglePasswordButton = document.getElementById(&#39;togglePassword&#39;);
            const eyeIcon = togglePasswordButton.querySelector(&#39;i&#39;);

            togglePasswordButton.addEventListener(&#39;click&#39;, function () {
                // Toggle the type attribute
                const type = passwordInput.getAttribute(&#39;type&#39;) === &#39;password&#39; ? &#39;text&#39; : &#39;password&#39;;
                passwordInput.setAttribute(&#39;type&#39;, type);

                // Toggle the eye icon class
                if (type === &#39;password&#39;) {
                    // If type is password, show the &#39;eye-slash&#39; icon
                    eyeIcon.classList.remove(&#39;fa-eye&#39;);
                    eyeIcon.classList.add(&#39;fa-eye-slash&#39;);
                } else {
                    // If type is text, show the &#39;eye&#39; icon
                    eyeIcon.classList.remove(&#39;fa-eye-slash&#39;);
                    eyeIcon.classList.add(&#39;fa-eye&#39;);
                }
            });
        });

        // --- NEW: Logic for the STUDENT password toggle ---
        const studentPasswordInput = document.getElementById(&#39;student_password&#39;);
        const toggleStudentPasswordButton = document.getElementById(&#39;toggleStudentPassword&#39;);

        // Check if the student elements actually exist before adding listener
        if (studentPasswordInput &amp;&amp; toggleStudentPasswordButton) {
            const studentEyeIcon = toggleStudentPasswordButton.querySelector(&#39;i&#39;);

            toggleStudentPasswordButton.addEventListener(&#39;click&#39;, function () {
                // Toggle the type attribute
                const type = studentPasswordInput.getAttribute(&#39;type&#39;) === &#39;password&#39; ? &#39;text&#39; : &#39;password&#39;;
                studentPasswordInput.setAttribute(&#39;type&#39;, type);

                // Toggle the eye icon class
                if (type === &#39;password&#39;) {
                    studentEyeIcon.classList.remove(&#39;fa-eye&#39;);
                    studentEyeIcon.classList.add(&#39;fa-eye-slash&#39;);
                } else {
                    studentEyeIcon.classList.remove(&#39;fa-eye-slash&#39;);
                    studentEyeIcon.classList.add(&#39;fa-eye&#39;);
                }
            });
        }

        // --- NEW: Logic for the MOBILE switch links ---
        const mobileSwitchToStaff = document.getElementById(&#39;mobile-switch-to-staff&#39;);
        const mobileSwitchToStudent = document.getElementById(&#39;mobile-switch-to-student&#39;);

        if (mobileSwitchToStaff) {
            mobileSwitchToStaff.addEventListener(&#39;click&#39;, (e) =&gt; {
                e.preventDefault();
                container.classList.add(&#34;right-panel-active&#34;);
            });
        }

        if (mobileSwitchToStudent) {
            mobileSwitchToStudent.addEventListener(&#39;click&#39;, (e) =&gt; {
                e.preventDefault();
                container.classList.remove(&#34;right-panel-active&#34;);
            });
        }
    </script>
